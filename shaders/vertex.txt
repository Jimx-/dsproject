#version 330 core
layout (location = 0) in vec4 aPosition;
layout (location = 1) in vec3 aNormal;
layout (location = 2) in vec2 aTexcoord;

uniform mat4 uModel;
uniform mat4 uView;
uniform mat4 uProjection;

out vec2 vTexcoord;

void main()
{
    gl_Position = uProjection * uView * uModel * aPosition;
    vTexcoord = aTexcoord;
}
